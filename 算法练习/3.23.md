1.字符串压缩
var compressString = function(S) {
  let s = S.replace(/(\w)\1*/g,(str,str1)=>{
    return str1+str.length
  })
  return s.length >= S.length ? S:s;
};
2.给定一个整数，将其转化为7进制，并以字符串形式输出。
var convertToBase7 = function(num) {
 if(num === 0)
        return '0';
    let s = num < 0 ? '-' : '';
    let res = [];
    let base = 7;
    num = Math.abs(num);
    while(num) {
        res.push(num % base);
        num = ~~(num / base);
    }
    return s + res.reverse().join('') ;
};
