merge = (A, m, B, n) => {
    if (B.length <= 0) {
        return
    }
    for (var i = m; i < m + n; i++) {
        A[i] = B.shift();
    }
    for (let i = m; i < A.length; i++) {
        let tmp = A[i]
        let j = i
        for (; j >= 0; j--) {
            if (tmp <= A[j - 1]) {
                A[j] = A[j - 1]
            } else {
                break
            }
        }
        A[j] = tmp
    }
};
