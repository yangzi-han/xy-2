reverseList = (head) => {
    function _reverse(head) {
      // 2. 从链表倒数第二个反向向前遍历
      // 2.1 倒数第二个节点的next的next置为当前节点
      // 2.2 倒数第三个节点的next的next置为当前节点
      // ...
      // 直到第一个节点
      if (head && head.next) {
        _reverse(head.next)
        head.next.next = head
        head.next = null
      } else { // 1. 找到链表尾巴，将其作为表头
          _reverse.head = head
      }
    }
    _reverse(head)
    return _reverse.head
  };
