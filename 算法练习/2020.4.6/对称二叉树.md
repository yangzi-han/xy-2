isSymmetric = (root) => {
  if (!root) {
    return true;
  }
  return isSameTree(root.left, root.right);
};

const isSameTree = (s, t) => {
  if (s === null) {
    return t === null;
  }
  if (t === null) {
    return s === null;
  }
  if (s.val !== t.val) {
    return false;
  }
  return isSameTree(s.left, t.right) && isSameTree(s.right, t.left);
};

mergeTrees = (t1, t2) => {
    let t;
    
    if(t1 && t2) {
        t = new TreeNode(t1.val + t2.val)
        t.left = mergeTrees(t1.left, t2.left);
        t.right = mergeTrees(t1.right, t2.right);
    } else {
        t = t1 || t2;
    }
    
    return t;
};

merge = (A, m, B, n) => {
    if (B.length <= 0) {
        return
    }
    for (var i = m; i < m + n; i++) {
        A[i] = B.shift();
    }
    for (let i = m; i < A.length; i++) {
        let tmp = A[i]
        let j = i
        for (; j >= 0; j--) {
            if (tmp <= A[j - 1]) {
                A[j] = A[j - 1]
            } else {
                break
            }
        }
        A[j] = tmp
    }
};
