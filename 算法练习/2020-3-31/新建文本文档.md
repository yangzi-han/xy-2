sortString = (s) => {
    s = s.split('').map(v => v.codePointAt())
    var res = []
    while(s.length) {
      let min = -Infinity
      let max = Infinity
      while(true) {
        min = Math.min(...s.filter(v => v > min))
        if(min === Infinity) {
          break;
        }
        res.push(min)
        s.splice(s.indexOf(min), 1)
      }
      while(true) {
        max = Math.max(...s.filter(v => v < max))
        if(max === -Infinity) {
          break;
        }
        res.push(max)
        s.splice(s.indexOf(max), 1)
      }
    }
    return res.map(v => String.fromCharCode(v)).join('')
  };
