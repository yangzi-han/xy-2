var isPalindrome = function(head) {
    if(head) {
        let arr = [head.val];
        while(head.next) {
            arr.push(head.next.val);
            head = head.next;
        }
        for(let i=0;i<arr.length/2;i++) {
            if(arr[i] != arr[arr.length-1-i]) {
               return false;
            }
        }
        return true;
    }
    return true;
};
