minArray = (numbers)=> {
  let left = 0
  let right = numbers.length-1
  let arr = numbers
  while (left < right) {
    let mid = (left + right)>>1
    if (arr[mid] > arr[right]) {
      left = mid+1
    }else if (arr[mid] == arr[right]){
      right--
    } else {
      right = mid
    }
  }
  return arr[left]
};
